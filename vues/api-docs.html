Les endpoints publics ci-dessous sont accessibles sans token. Certains endpoints internes affiches dans le Swagger necessitent une authentification Bearer. Voici des exemples prets a l'emploi pour l'API publique en JavaScript (discord.js v14) et Python (discord.py). 
const BASE = 'https://zenkai-police.tech/api/v1';

async function listMembres(rang) {
  const url = rang
    ? `${BASE}/membres?rang=${encodeURIComponent(rang)}`
    : `${BASE}/membres`;

  const response = await fetch(url);
  if (!response.ok) throw new Error('Impossible de lire le registre');
  return response.json();
}

async function ficheMembre(pseudoHRP) {
  const response = await fetch(`${BASE}/membres/${encodeURIComponent(pseudoHRP)}`);
  if (!response.ok) throw new Error(`Membre introuvable : ${pseudoHRP}`);
  return response.json();
}

async function presencePublique() {
  const response = await fetch(`${BASE}/presence`);
  return response.json();
}

}

import aiohttp
import urllib.parse

BASE = "https://zenkai-police.tech/api/v1"

async def list_membres(rang=None):
    url = f"{BASE}/membres"
    if rang:
        url += f"?rang={urllib.parse.quote(rang)}"

    async with aiohttp.ClientSession() as session:
        async with session.get(url) as response:
            response.raise_for_status()
            return await response.json()

async def fiche_membre(pseudo_hrp):
    url = f"{BASE}/membres/{urllib.parse.quote(pseudo_hrp)}"
    async with aiohttp.ClientSession() as session:
        async with session.get(url) as response:
            response.raise_for_status()
            return await response.json()
